Um später die Reihenfolge der einzelnen Befragungen eindeutig bestimmen zu können, berechnen wir eine reihenfolge_id. Diese ID wird anhand der Abbildung f beschrieben, die ein x und ein a innerhalb der natürlichen Zahlen abbildet. Dabei ist x der Tag und a die Tageszeit.

\[f: \mathbb{N} \to \mathbb{N}, x, a \mapsto (x - 1) \times 3 + a \forall x \in [1;7] \land a \in [1;3]\]

#erstelle occasion_id


#vormittag
vormittag_raw <- vormittag_raw %>%
  mutate(
    scheduled_converted = ymd_hms(scheduled)
  )

vormittag_raw <- vormittag_raw %>%
  group_by(personalParticipantCode) %>%

    studientag = as.integer(
      as.Date(

#erstelle variable lag
merged_dataset <- merged_dataset %>%
  arrange(personalParticipantCode, reihenfolge_id) %>%
  group_by(personalParticipantCode) %>%
  mutate(
    habit_lag = if_else(
      reihenfolge_id - lag(reihenfolge_id) == 1,
      lag(index_habit),
      NA_real_
    )
  ) %>%
  ungroup()

#rechne Modell für H1
model_h1 <- lmer(
  index_habit ~ habit_lag + index_beduerfnis + (1 | personalParticipantCode),
